# Declaring high-DPI awareness in a .NET 6/7 WinForms app.

# --- Option A: the template default (Program.cs) ---
# ApplicationConfiguration.Initialize() applies the configured HighDpiMode.

# --- Option B: set the mode explicitly in Program.cs ---
Application.SetHighDpiMode(HighDpiMode.PerMonitorV2);
Application.EnableVisualStyles();
Application.SetCompatibleTextRenderingDefault(false);
Application.Run(new MainForm());

# --- Option C: in the .csproj ---
# <PropertyGroup>
#   <ApplicationHighDpiMode>PerMonitorV2</ApplicationHighDpiMode>
# </PropertyGroup>

# HighDpiMode values:
#   DpiUnaware          - Windows bitmap-stretches (blurry). Avoid.
#   SystemAware         - sharp at the primary monitor's scale only.
#   PerMonitor          - rescales per monitor (legacy).
#   PerMonitorV2        - BEST: correct rescaling across mixed-DPI monitors.

# On the form, scale with the layout system, not pixels:
#   AutoScaleMode = AutoScaleMode.Dpi;   // (or Font)

# In custom GDI+ drawing, scale by DPI:
#   float scale = DeviceDpi / 96f;
#   g.FillRectangle(brush, 0, 0, 100 * scale, 40 * scale);
